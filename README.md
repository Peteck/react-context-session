## Install

```bash
npm install --save react-context-session
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-context-session'
import 'react-context-session/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
